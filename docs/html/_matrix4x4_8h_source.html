176  static Matrix4x4 rotationAround(const Vector3 &axis, double angle)
177  {
178  Vector3 axisNormalized = axis.normalized();
179  double q0 = cos(angle/2);
180  double q1 = sin(angle/2) * axisNormalized.x();
181  double q2 = sin(angle/2) * axisNormalized.y();
182  double q3 = sin(angle/2) * axisNormalized.z();
183 
184  Matrix4x4 result;
185  result(0,0) = q0*q0 + q1*q1 - q2*q2 - q3*q3;
186  result(0,1) = 2*(q1*q2 - q0*q3);
187  result(0,2) = 2*(q1*q3 + q0*q2);
188  result(1,0) = 2*(q2*q1 + q0*q3);
189  result(1,1) = (q0*q0 - q1*q1 + q2*q2 - q3*q3);
190  result(1,2) = 2*(q2*q3 - q0*q1);
191  result(2,0) = 2*(q3*q1 - q0*q2);
192  result(2,1) = 2*(q3*q2 + q0*q1);
193  result(2,2) = (q0*q0 - q1*q1 - q2*q2 + q3*q3);
194  return result;
195  }
