48  bool render(const Scene &scene, Image<unsigned char, 3> &image) override
49  {
50  auto camera = scene.camera();
51  if (camera == nullptr)
52  {
53  return false;
54  }
55 
56  // For now only support orthographic projection
57  std::shared_ptr<OrthographicCamera> cameraOrtho = std::dynamic_pointer_cast<OrthographicCamera>(camera);
58  if (cameraOrtho == nullptr)
59  {
60  return false;
61  }
62 
63  // 1. Construct view port matrix based on image resolution.
64  double w = static_cast<double>(image.width());
65  double h = static_cast<double>(image.height());
66 
67  geom3d::Matrix4x4 matrixViewport({ { w/2, 0 , 0, (w-1)/2 },
68  { 0 , h/2, 0, (h-1)/2 },
69  { 0 , 0 , 1, 0 },
70  { 0 , 0 , 0, 1 } });
71 
72 
73  // 2. Construct orthographic projection matrix based on canonical view volume.
74  const double n = -cameraOrtho->near();
75  const double f = -cameraOrtho->far();
76  const double r = cameraOrtho->size() * 0.5 * w/h; // w/h = aspect ratio
77  const double l = -r;
78  const double t = cameraOrtho->size() * 0.5;
79  const double b = -t;
80 
81  geom3d::Matrix4x4 matrixOrtho({ { 2/(r-l), 0 , 0 , -(r+l)/(r-l) },
82  { 0 , 2/(t-b), 0 , -(t+b)/(t-b) },
83  { 0 , 0 , 2/(n-f), -(n+f)/(n-f) },
84  { 0 , 0 , 0 , 1 } });
85 
86  // Iterate objects in scene
87  for (auto renderObject : scene.renderObjects())
88  {
89  // Cast to mesh
90  std::shared_ptr<Mesh> mesh = std::dynamic_pointer_cast<Mesh>(renderObject);
91  if (mesh)
92  {
93  // 3. Construct camera transformation matrix (World space -> Camera space)
94  geom3d::Matrix4x4 matrixCamera = cameraOrtho->transform().matrix().inverse();
95 
96  // 4. Construct world transformation matrix (object space -> world space)
97  geom3d::Matrix4x4 matrixObject = mesh->transform().matrix();
98 
99  // Combine matrices to form a matrix that projects a point from local
100  // object space to screen space.
101  geom3d::Matrix4x4 matrixPipeline = matrixViewport * matrixOrtho *
102  matrixCamera * matrixObject;
103 
104  // Iterate edges in object
105  for (auto edge : mesh->edges())
106  {
107  // Get vertices of edge
108  geom3d::Point3 v1 = mesh->vertex(edge[0]);
109  geom3d::Point3 v2 = mesh->vertex(edge[1]);
110 
111  // Project vertices to screen space
112  geom3d::Point3 v1Proj = matrixPipeline * v1;
113  geom3d::Point3 v2Proj = matrixPipeline * v2;
114 
115  int x1 = static_cast<int>(v1Proj.x());
116  int y1 = static_cast<int>(v1Proj.y());
117  int x2 = static_cast<int>(v2Proj.x());
118  int y2 = static_cast<int>(v2Proj.y());
119 
120  gfx2d::Drawing::drawLine(image, {x1, y1}, {x2, y2}, {255, 255, 255});
121  }
122  }
123  }
124  }
125 
126 private:
127  int m_edgeThickness;
128 };
