56  {
57  const std::vector<std::array<double, 2>> &P = m_points;
58  const int N = degree();
59 
60  // Check for at least two anchor points

64  }
65 
66  double x = 0, y = 0;
67  for(int i = 0; i <= N; i++)
68  {
69  x += binomialCoefficient(N, i) * pow(1 - t, N - i) * pow(t, i) * P[i][0];
70  y += binomialCoefficient(N, i) * pow(1 - t, N - i) * pow(t, i) * P[i][1];
71  }
